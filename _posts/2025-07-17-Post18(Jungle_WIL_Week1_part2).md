li = [[1, 5, 3], [2, 5, 7], [5, 3, 5]]
check = [False] * 3
m = sys.maxsize

def backtracking(row, score):
  global m

  if row == 3:
    m = min(m, score)
    return

  for i in range(3):
    if not check[i]:
      check[i] = True
      backtracking(row + 1, score + li[row][i])
      check[i] = False

      backtracking(0, 0)
print(m)  # 8
```

def insertion_sort(num_list):

  for i in range(1, len(num_list)):
    key = num_list[i]
    j = i - 1

    while j >= 0 and num_list[j] > key:
      num_list[j + 1] = num_list[j]
      j -= 1

    num_list[j + 1] = key

  return num_list
```
